"""Version-safe PaddleOCR wrapper."""

from __future__ import annotations

import inspect
import os
from typing import Any

import numpy as np
import cv2

from app.core.config import ModelSettings
from app.core.logging import get_logger
from app.ocr.types import OCRLine

logger = get_logger(__name__)

os.environ.setdefault("PADDLE_PDX_DISABLE_MODEL_SOURCE_CHECK", "True")
os.environ.setdefault("FLAGS_use_mkldnn", "0")

def _patch_paddlex_device_if_needed() -> None:
    try:
        import paddle  # noqa: F401
        if hasattr(paddle, "device"):
            return
        import paddlex.utils.device as pdx_device
        def _cpu_only() -> str:
            return "cpu"
        pdx_device.get_default_device = _cpu_only
    except ImportError:
        pass

try:
    _patch_paddlex_device_if_needed()
    from paddleocr import PaddleOCR  # type: ignore
except ImportError:
    PaddleOCR = None

try:
    from doctr.models import ocr_predictor  # type: ignore
except ImportError:
    ocr_predictor = None


class PaddleEngine:
    def __init__(self, config: ModelSettings):
        if PaddleOCR is None:
            raise ImportError("PaddleOCR is not installed. Install backend requirements first.")
        signature = inspect.signature(PaddleOCR.__init__)
        accepted = set(signature.parameters.keys())
        paddle_args: dict[str, Any] = {}
        if "lang" in accepted:
            paddle_args["lang"] = config.paddle_lang
        if "use_textline_orientation" in accepted:
            paddle_args["use_textline_orientation"] = config.use_textline_orientation
        elif "use_angle_cls" in accepted:
            paddle_args["use_angle_cls"] = config.use_textline_orientation
        if "show_log" in accepted:
            paddle_args["show_log"] = False
        if "use_gpu" in accepted:
            paddle_args["use_gpu"] = False
        if "enable_mkldnn" in accepted:
            paddle_args["enable_mkldnn"] = False
        logger.info("Loading PaddleOCR with args: %s", paddle_args)
        self.ocr = PaddleOCR(**paddle_args)
        self._doctr_predictor = None

    def _get_doctr_predictor(self):
        if ocr_predictor is None:
            raise RuntimeError("DocTR is not installed, so OCR fallback is unavailable.")
        if self._doctr_predictor is None:
            logger.warning("Falling back to DocTR OCR because PaddleOCR failed at runtime.")
            self._doctr_predictor = ocr_predictor(pretrained=True)
        return self._doctr_predictor

    def _call_ocr(self, img_bgr: np.ndarray):
        try:
            return self.ocr.ocr(img_bgr, cls=True)
        except TypeError:
            return self.ocr.ocr(img_bgr)

    def _run_doctr_fallback(self, img_bgr: np.ndarray) -> list[OCRLine]:
        predictor = self._get_doctr_predictor()
        rgb = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2RGB)
        height, width = rgb.shape[:2]
        result = predictor([rgb])
        lines: list[OCRLine] = []
        for page in result.pages:
            for block in page.blocks:
                for line in block.lines:
                    if not line.words:
                        continue
                    points = []
                    tokens = []
                    for word in line.words:
                        ((x0, y0), (x1, y1)) = word.geometry
                        points.extend([(int(x0 * width), int(y0 * height)), (int(x1 * width), int(y1 * height))])
                        if word.value:
                            tokens.append(word.value)
                    if not points:
                        continue
                    xs = [point[0] for point in points]
                    ys = [point[1] for point in points]
                    text = " ".join(tokens).strip()
                    lines.append(OCRLine(bbox=(min(xs), min(ys), max(xs), max(ys)), raw_text=text, confidence=0.65, source="doctr"))
        return lines

    def run(self, img_bgr: np.ndarray) -> list[OCRLine]:
        try:
            result = self._call_ocr(img_bgr)
        except Exception as exc:
            logger.warning("PaddleOCR failed, switching to DocTR fallback: %s", exc)
            lines = self._run_doctr_fallback(img_bgr)
            lines.sort(key=lambda line: (line.bbox[1], line.bbox[0]))
            return lines
        lines: list[OCRLine] = []
        if not result:
            return lines
        first_item = result[0]
        if isinstance(first_item, list):
            if not first_item:
                return lines
            for item in first_item:
                points, (text, conf) = item
                xs = [int(point[0]) for point in points]
                ys = [int(point[1]) for point in points]
                lines.append(OCRLine(bbox=(min(xs), min(ys), max(xs), max(ys)), raw_text=text or "", confidence=float(conf), source="paddle"))
        else:
            for item in result:
                data = getattr(item, "res", item)
                if hasattr(data, "to_dict"):
                    data = data.to_dict()
                if not isinstance(data, dict):
                    continue
                texts = data.get("rec_texts") or []
                scores = data.get("rec_scores") or []
                boxes = data.get("rec_boxes") or []
                polys = data.get("dt_polys") or []
                for index, text in enumerate(texts):
                    conf = float(scores[index]) if index < len(scores) else 0.0
                    box = boxes[index] if index < len(boxes) else None
                    poly = polys[index] if index < len(polys) else None
                    if box and len(box) == 4:
                        x0, y0, x1, y1 = [int(value) for value in box]
                    elif poly is not None:
                        xs = [int(point[0]) for point in poly]
                        ys = [int(point[1]) for point in poly]
                        x0, y0, x1, y1 = min(xs), min(ys), max(xs), max(ys)
                    else:
                        continue
                    lines.append(OCRLine(bbox=(x0, y0, x1, y1), raw_text=text or "", confidence=conf, source="paddle"))
        lines.sort(key=lambda line: (line.bbox[1], line.bbox[0]))
        return lines

"""TrOCR engine for difficult handwritten crops."""

from __future__ import annotations

import cv2
import numpy as np
import torch
from PIL import Image
from transformers import TrOCRProcessor, VisionEncoderDecoderModel

from app.core.config import ModelSettings
from app.core.logging import get_logger
from app.utils.diffing import normalize_spaces

logger = get_logger(__name__)


class TrOCREngine:
    def __init__(self, config: ModelSettings):
        logger.info("Loading TrOCR fallback model: %s", config.trocr_model_name)
        self.processor = TrOCRProcessor.from_pretrained(config.trocr_model_name, use_fast=False)
        self.model = VisionEncoderDecoderModel.from_pretrained(config.trocr_model_name)
        self.model.to("cpu")
        self.model.eval()
        self.max_new_tokens = config.trocr_max_new_tokens
        self.num_beams = config.trocr_num_beams

    def recognize_crop(self, crop_bgr: np.ndarray) -> str:
        crop_rgb = cv2.cvtColor(crop_bgr, cv2.COLOR_BGR2RGB)
        pil_image = Image.fromarray(crop_rgb)
        pixel_values = self.processor(images=pil_image, return_tensors="pt").pixel_values
        with torch.no_grad():
            generated_ids = self.model.generate(
                pixel_values,
                max_new_tokens=self.max_new_tokens,
                num_beams=self.num_beams,
                early_stopping=True,
            )
        text = self.processor.batch_decode(generated_ids, skip_special_tokens=True)[0]
        return normalize_spaces(text)

def refine_spelling(text: str) -> str:
    if not text or not text.strip():
        return text
    result = text
    for wrong, right in SPELLING_MAP:
        pattern = re.compile(r"\b" + re.escape(wrong) + r"\b", re.IGNORECASE)
        def repl(m):
            orig = m.group(0)
            if orig and orig[0].isupper():
                return right.capitalize() if len(right) > 1 else right.upper()
            return right
        result = pattern.sub(repl, result)
    return normalize_spaces(result)

"""Fast local lexical cleanup."""

from __future__ import annotations

import re
from dataclasses import dataclass

from rapidfuzz import fuzz, process

from app.utils.diffing import normalize_spaces


DEFAULT_LEXICON = [
    "student", "teacher", "school", "class", "handwriting", "paragraph",
    "sentence", "word", "reading", "writing", "president", "commonwealth",
    "presence", "revolution", "counter", "after", "threatened", "occasion",
    "worthy", "states", "these",
]


@dataclass
class LexicalCorrection:
    text: str
    changed_tokens: list[dict[str, str]]


class LexicalCorrector:
    def __init__(self, lexicon: list[str] | None = None):
        self.lexicon = lexicon or DEFAULT_LEXICON

    def _cleanup_punctuation(self, text: str) -> str:
        text = text.replace(" . ", " ").replace(" .", " ")
        text = re.sub(r"\s+([,.;:!?])", r"\1", text)
        text = re.sub(r"([,.;:!?])([A-Za-z])", r"\1 \2", text)
        text = re.sub(r"[^A-Za-z0-9\s.,!?;:'\"()\-/]", "", text)
        return normalize_spaces(text)

    def _token_rescue(self, token: str) -> str:
        lower = token.lower()
        if len(lower) < 4 or not lower.isalpha():
            return token
        match = process.extractOne(lower, self.lexicon, scorer=fuzz.ratio, score_cutoff=90)
        if not match:
            return token
        rescued = match[0]
        if token[:1].isupper():
            return rescued.capitalize()
        return rescued

    def correct(self, text: str) -> LexicalCorrection:
        cleaned = self._cleanup_punctuation(text)
        changed_tokens: list[dict[str, str]] = []
        tokens = re.findall(r"\w+|[^\w\s]", cleaned, flags=re.UNICODE)
        rescued: list[str] = []
        for token in tokens:
            replacement = self._token_rescue(token)
            if replacement != token:
                changed_tokens.append({"from": token, "to": replacement})
            rescued.append(replacement)
        rebuilt = normalize_spaces(" ".join(rescued).replace(" ,", ",").replace(" .", "."))
        return LexicalCorrection(text=rebuilt, changed_tokens=changed_tokens)

"""Higher-level correction orchestration."""

from __future__ import annotations

import re
from difflib import SequenceMatcher

from app.correction.byt5_corrector import ByT5Corrector
from app.correction.lexical import LexicalCorrector
from app.correction.spelling_refinement import refine_spelling
from app.ocr.types import OCRLine
from app.utils.diffing import correction_ratio, normalize_spaces, weird_char_ratio


class ParagraphCorrector:
    def __init__(self, lexical_corrector: LexicalCorrector, byt5_corrector: ByT5Corrector | None = None):
        self.lexical_corrector = lexical_corrector
        self.byt5_corrector = byt5_corrector

    @staticmethod
    def _token_overlap_ratio(before: str, after: str) -> float:
        before_tokens = re.findall(r"[A-Za-z']+", (before or "").lower())
        after_tokens = re.findall(r"[A-Za-z']+", (after or "").lower())
        if not before_tokens or not after_tokens:
            return 0.0
        matcher = SequenceMatcher(a=before_tokens, b=after_tokens)
        matched = sum(i2 - i1 for tag, i1, i2, _, _ in matcher.get_opcodes() if tag == "equal")
        return matched / max(len(before_tokens), len(after_tokens), 1)

    def _accept_candidate(self, source_text: str, candidate_text: str) -> str:
        source = normalize_spaces(source_text)
        candidate = normalize_spaces(candidate_text)
        if not source or not candidate:
            return source
        source_noise = weird_char_ratio(source)
        if source_noise > 0.20:
            return source
        overlap = self._token_overlap_ratio(source, candidate)
        change = correction_ratio(source, candidate)
        candidate_noise = weird_char_ratio(candidate)
        lowered = candidate.lower()
        if "repair ocr noise" in lowered or "correct ocr" in lowered:
            return source
        if overlap < 0.40:
            return source
        if change > 0.55 and overlap < 0.65:
            return source
        if candidate_noise > source_noise + 0.03:
            return source
        return candidate

    def _repair_with_context(self, paragraph: str) -> str:
        if not self.byt5_corrector:
            return paragraph
        candidate = self.byt5_corrector.correct(paragraph)
        accepted = self._accept_candidate(paragraph, candidate)
        if accepted != paragraph:
            return accepted
        if not hasattr(self.byt5_corrector, "chunk_text"):
            return paragraph
        chunks = self.byt5_corrector.chunk_text(paragraph, max_chars=140)
        if len(chunks) <= 1:
            return paragraph
        repaired_chunks = [self._accept_candidate(chunk, self.byt5_corrector.correct(chunk)) for chunk in chunks]
        return normalize_spaces(" ".join(repaired_chunks))

    def correct_line(self, line: OCRLine) -> OCRLine:
        base_text = normalize_spaces(line.merged_text or line.raw_text)
        if not base_text or weird_char_ratio(base_text) > 0.22:
            line.corrected_text = base_text
            line.uncertainty_score = min(1.0, max(1.0 - line.confidence, 0.85))
            return line
        lexical = self.lexical_corrector.correct(base_text)
        corrected = lexical.text
        line.corrected_text = normalize_spaces(corrected)
        line.uncertainty_score = min(1.0, max(1.0 - line.confidence, correction_ratio(base_text, line.corrected_text)))
        return line

    def correct_paragraph(self, lines: list[OCRLine]) -> tuple[str, str, str, str]:
        corrected_lines = [self.correct_line(line).corrected_text or "" for line in lines]
        paragraph = normalize_spaces(" ".join(corrected_lines))
        layer1 = paragraph
        line_uncertainty = [line.uncertainty_score for line in lines]
        avg_uncertainty = sum(line_uncertainty) / max(1, len(line_uncertainty))
        suspicious_count = sum(1 for line in lines if line.suspicious)
        if (self.byt5_corrector and len(paragraph) > 40 and len(paragraph) <= 220 and len(lines) <= 5
            and avg_uncertainty < 0.65 and weird_char_ratio(paragraph) < 0.12 and suspicious_count <= 2):
            paragraph = self._repair_with_context(paragraph)
        layer2 = paragraph
        paragraph = refine_spelling(paragraph)
        layer3 = normalize_spaces(paragraph)
        return (layer3, layer1, layer2, layer3)

"""Local ByT5 correction wrapper."""

from __future__ import annotations

import re

import torch
from transformers import AutoModelForSeq2SeqLM, AutoTokenizer

from app.core.config import ModelSettings
from app.core.logging import get_logger
from app.utils.diffing import normalize_spaces

logger = get_logger(__name__)


class ByT5Corrector:
    def __init__(self, config: ModelSettings):
        logger.info("Loading local ByT5 correction model: %s", config.byt5_model_name)
        self.tokenizer = AutoTokenizer.from_pretrained(config.byt5_model_name)
        self.model = AutoModelForSeq2SeqLM.from_pretrained(config.byt5_model_name)
        self.model.to("cpu")
        self.model.eval()
        self.max_length = config.correction_max_length
        self.num_beams = config.correction_num_beams
        self.prefix = (
            "Repair OCR noise in this paragraph. Preserve names, topic, sentence order, and factual meaning. "
            "Only fix likely recognition errors, spacing, punctuation, and obvious spelling mistakes supported by context. "
            "Do not add new facts or rewrite the paragraph into a different message. Return only the corrected paragraph: "
        )
        self.echo_markers = [
            "Correct OCR and dyslexia-related spelling errors while preserving meaning:",
            "Repair OCR noise in this paragraph while preserving meaning, names, and the original order of ideas.",
            "Do not invent new facts. Return only the corrected paragraph:",
        ]

    def chunk_text(self, text: str, max_chars: int = 220) -> list[str]:
        text = normalize_spaces(text)
        if len(text) <= max_chars:
            return [text] if text else []
        sentences = re.split(r"(?<=[.!?])\s+", text)
        chunks: list[str] = []
        current = ""
        for sentence in sentences:
            proposal = normalize_spaces(f"{current} {sentence}")
            if current and len(proposal) > max_chars:
                chunks.append(current)
                current = sentence
            else:
                current = proposal
        if current:
            chunks.append(current)
        return chunks

    def _correct_chunk(self, text: str) -> str:
        if not text.strip():
            return text
        prompt = self.prefix + text
        inputs = self.tokenizer(prompt, return_tensors="pt", truncation=True)
        with torch.no_grad():
            outputs = self.model.generate(**inputs, max_length=self.max_length, num_beams=self.num_beams, early_stopping=True)
        decoded = normalize_spaces(self.tokenizer.decode(outputs[0], skip_special_tokens=True))
        for marker in self.echo_markers:
            decoded = normalize_spaces(decoded.replace(marker, " "))
        if decoded.lower().startswith("repair ocr noise") or decoded.lower().startswith("correct ocr"):
            return text
        return decoded or text

    def correct(self, text: str) -> str:
        chunks = self.chunk_text(text)
        return normalize_spaces(" ".join(self._correct_chunk(chunk) for chunk in chunks))

"""Conservative OCR candidate selection."""

from __future__ import annotations

import re

from app.utils.diffing import digit_ratio, normalize_spaces, weird_char_ratio


class FusionEngine:
    COMMON_WORDS = {
        "the", "and", "is", "are", "was", "were", "this", "that", "with", "for",
        "have", "has", "you", "your", "from", "will", "can", "not", "school",
        "student", "page", "word", "sentence", "paragraph", "teacher", "reading",
        "writing", "handwriting", "hope", "hear", "soon",
    }

    @staticmethod
    def language_plausibility_score(text: str) -> float:
        text = normalize_spaces(text)
        if not text:
            return -1e9
        weird_penalty = weird_char_ratio(text) * 6.0
        digit_penalty = max(0.0, digit_ratio(text) - 0.25) * 4.0
        tokens = re.findall(r"[A-Za-z']+", text.lower())
        if not tokens:
            token_score = -2.0
        else:
            common_hits = sum(token in FusionEngine.COMMON_WORDS for token in tokens)
            avg_token_len = sum(len(token) for token in tokens) / max(1, len(tokens))
            token_score = (common_hits / max(1, len(tokens))) * 2.5
            token_score += min(avg_token_len / 6.0, 1.0)
        punctuation_balance = 0.0
        if text.count("(") == text.count(")"):
            punctuation_balance += 0.2
        if text.count('"') % 2 == 0:
            punctuation_balance += 0.1
        return token_score + punctuation_balance - weird_penalty - digit_penalty

    @staticmethod
    def choose_text(primary_text: str, fallback_text: str) -> str:
        primary = normalize_spaces(primary_text)
        fallback = normalize_spaces(fallback_text)
        if not fallback:
            return primary
        if not primary:
            return fallback
        primary_score = FusionEngine.language_plausibility_score(primary)
        fallback_score = FusionEngine.language_plausibility_score(fallback)
        if fallback_score > primary_score + 0.2:
            return fallback
        return primary
